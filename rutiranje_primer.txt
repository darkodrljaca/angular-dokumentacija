// ako je id broj popuni formu:
  ngOnInit() {
    let id: number = Number(this.route.snapshot.params['id']);    
    if (id) {      
      let obs: Observable<Wine> = this.wineService.get(id);
      obs.subscribe({
        next: (odgovor: any) => {          
          this.wineForm.patchValue(odgovor);
        },
        error: (err: any) => {
          alert(err.statusText);        
        }
      });            
    }
  }

// dodaj ili azuriraj vino:
  onSubmit(): void {
    this.wine = new Wine(this.wineForm.value);    
    let id: number = Number(this.route.snapshot.params['id']);
    if (id) {
      // update wine            
      this.wine._id = id;      
      let obs: Observable<Wine> = this.wineService.update(this.wine, id);
      obs.subscribe({
        next: (odgovor: any) => {          
          this.router.navigate(['wines']);
        },
        error: (err: any) => {
          console.log(err.statusText);          
        }
      });
    } else {
      // add wine
      let obs: Observable<Wine> = this.wineService.add(this.wine);
      obs.subscribe({
        next: (odgovor: any) => {                    
          this.router.navigate(['wines']);
        },
        error: (err: any) => {
          console.log(err.statusText);          
        }
      });
    }        
  }
